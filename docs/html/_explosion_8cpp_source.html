207 void Explosion::getPointInSphere(glm::vec3* point, float radius) const
208 {
209  // http://math.stackexchange.com/questions/87230/picking-random-points-in-the-volume-of-sphere-with-uniform-probability

213  float theta = randf(0.f, static_cast<float>(tau));
214  float v = randf(0.f, 1.f);
