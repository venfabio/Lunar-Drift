230 void ParticleSystem::getPointInSphere(glm::vec3* point, float radius) const
231 {
232  // http://math.stackexchange.com/questions/87230/picking-random-points-in-the-volume-of-sphere-with-uniform-probability
233  const double pi = 3.14159265359;
234  const double tau = 6.28318530718;
235 
236  float theta = randf(0.f, static_cast<float>(tau));
237  float v = randf(0.f, 1.f);
238  float phi = acos(2 * v - 1);
239  float r = radius * pow(randf(0.f, 1.f), 1.f / 3.f);
240 
241  *point = glm::vec3(
242  r * cos(theta) * sin(phi),
243  r * sin(theta) * sin(phi),
244  r * cos(phi));
